<script>
  // ====== CONFIGURAÇÕES ======
  const CHECKOUT_URL = 'https://seu-checkout-aqui.com'; // <-- COLE A URL DO SEU CHECKOUT
  const INITIAL_BUYERS = 1347; // contador simples ilustrativo

  // ====== Estados ======
  const steps = ['intro','quiz','result','plan','proof','offer'];
  const data = { answers: {}, score: 85, goal: 'R$2.000+' };

  // ====== Elementos ======
  const el = id => document.getElementById(id);
  const quizEl = el('quiz');
  const bar = el('bar');
  const progress = document.getElementById('progress');

  // ====== Pixel helpers (seguros) ======
  function track(event, params={}){ if(window.fbq){ try{ fbq('track', event, params);}catch(e){} } }

  // ====== Util ======
  function show(id){ steps.forEach(s=>{ const sec=document.getElementById(s==='intro'? 'step-intro':s); if(sec) sec.style.display = ( (s==='intro'? 'step-intro':s)===id ? 'block':'none');}); }
  function setProgress(i,total){ bar.style.width = Math.round((i/total)*100)+'%'; el('progressLabel').textContent = `Pergunta ${i} de ${total}`; }

  // ====== Perguntas ======
  const questions = [
    {
      key:'experiencia',
      title:'Qual é o seu nível de experiência na cozinha?',
      options:[
        '👩\u200d🔰 Nenhum, mas adoro aprender coisas novas.',
        '🍬 Já faço doces de vez em quando.',
        '👩\u200d🍳 Tenho experiência e quero dominar os panetones gourmet.'
      ]
    },
    {
      key:'meta',
      title:'Quanto você gostaria de faturar com panetones neste Natal?',
      options:[
        '💰 Até R$2.000,00','💵 Entre R$2.000 e R$5.000','💸 Mais de R$5.000','🧁 Ainda não pensei nisso, mas adoraria ter um lucro extra.'
      ]
    },
    {
      key:'bloqueio',
      title:'O que mais te impede de começar algo novo para ganhar uma renda extra?',
      options:[
        '⏳ Falta de tempo','😟 Medo de não dar certo','💸 Falta de dinheiro para investir','🤔 Não sei por onde começar'
      ]
    }
  ];

  // ====== Render Quiz ======
  let qIndex = 0;
  function renderQuestion(){
    const total = questions.length;
    const q = questions[qIndex];
    quizEl.innerHTML = `
      <h2>${q.title}</h2>
      <div class="answers">
        ${q.options.map((op,i)=>`<button class="answer" data-i="${i}">${op}</button>`).join('')}
      </div>`;
    quizEl.querySelectorAll('.answer').forEach(btn=>{
      btn.addEventListener('click', ()=>{
        data.answers[q.key] = btn.textContent.trim();
        if(q.key==='meta'){ data.goal = btn.textContent.replace(/[^0-9R$.,+]/g,'') || 'R$2.000+'; }
        qIndex++;
        if(qIndex<total){ setProgress(qIndex+1,total); renderQuestion(); }
        else { finishQuiz(); }
      });
    });
  }

  function finishQuiz(){
    show('result');
    track('ViewContent', {content_name:'Quiz Result'});
    el('goal').textContent = data.goal || 'R$2.000+';
    // score simples só para feedback
    let score = 85;
    if(data.answers.experiencia && data.answers.experiencia.includes('Nenhum')) score = 78;
    if(data.answers.meta && data.answers.meta.includes('Mais de')) score = 90;
    el('score').textContent = score + '/100';
  }

  // ====== Timer oferta ======
  let timerInterval;
  function startTimer(minutes=8){
    let secs = minutes*60; clearInterval(timerInterval);
    timerInterval = setInterval(()=>{
      const m = String(Math.floor(secs/60)).padStart(2,'0');
      const s = String(secs%60).padStart(2,'0');
      el('timer').textContent = `${m}:${s}`;
      if(--secs<0){ clearInterval(timerInterval); el('timer').textContent='00:00'; }
    },1000);
  }

  // ====== Eventos UI ======
  document.getElementById('startBtn').addEventListener('click',()=>{
    show('quiz'); progress.style.display='block'; setProgress(1,questions.length); renderQuestion();
    track('Lead', {source:'Start Quiz'});
  });

  document.getElementById('seePlan').addEventListener('click',()=>{
    show('plan');
    track('AddToCart', {step:'See Plan'});
  });

  document.getElementById('gotoOffer').addEventListener('click',()=>{ openOffer(); });
  document.getElementById('gotoProof').addEventListener('click',()=>{ show('proof'); });
  document.getElementById('toOffer2').addEventListener('click',()=>{ openOffer(); });

  function openOffer(){
    show('offer'); startTimer(8); el('sticky').style.display='block';
    track('AddToCart', {step:'Offer View'});
  }

  // Checkout links
  ['checkoutBtn','stickyCheckout'].forEach(id=>{
    const a = el(id); a.href = CHECKOUT_URL; a.addEventListener('click', ()=> track('InitiateCheckout')); });

  // Data e rodapé
  const today = new Date();
  el('today').textContent = today.toLocaleDateString('pt-BR');
  el('year').textContent = today.getFullYear();

  // contador fake de compradores (ilustrativo)
  let buyers = INITIAL_BUYERS; el('buyers').textContent = buyers.toLocaleString('pt-BR');
</script>
